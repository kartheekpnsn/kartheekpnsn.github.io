def process(test_str):
	lines = [i.strip() for i in test_str.splitlines()]
	words = [i.strip() for i in test_str.split()]
	two_words = [i + " " + j for i, j in zip(words[:-1],words[1:])]
	three_words = [i + " " + j + " " + k for i, j, k in zip(words[:-2], words[1:-1], words[2:])]
	print lines
	print words
	print two_words
	print three_words
	
text = "KFC Restaurant\nneed some english\nsome more english\n blah blah time"

process(text)
