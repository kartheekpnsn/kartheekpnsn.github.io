from itertools import combinations

# input - taken as dictionary to maintain the order
ip = {1:70, 2:30, 3:60, 4:80}

N = len(ip)

# # output
main_op = list()

for i in range(0, N + 1):
	print "Iteration: " + str(i + 1)
	all_value_comb = combinations(ip.values(), i)
	all_key_comb = list(combinations(ip.keys(), i))
	op = list()
	for each_key_comb, each_value_comb in zip(all_key_comb, all_value_comb):
		keys = list(each_key_comb)
		# to get the other elements that are not subtracted by 100
		missedKeys = list(set(ip.keys()) - set(keys))
		# to get the final order back to input
		finalKeys = missedKeys + keys
		# got the missed elements that are not subtracted by 100 using missedKeys
		missedElements = [ip[miss] for miss in list(set(ip.keys()) - set(keys))]
		# Elements that are to be subtracted by 100
		processedElements = [100 - each_value_ele for each_value_ele in each_value_comb]
		# Combine missed Elements and Elements that are subtracted by 100
		finalElements = missedElements + processedElements
		# Retain the order (as per given in the input)..this is achieved through the keys (finalKeys)
		print dict(zip(finalKeys, finalElements)).values()
		op.append(dict(zip(finalKeys, finalElements)).values())
	print "\n"
	main_op.append(op)
print main_op
